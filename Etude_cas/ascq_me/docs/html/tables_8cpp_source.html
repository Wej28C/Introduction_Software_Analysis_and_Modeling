104  "THE AMINO ACID IS NOT IN [A-Z] IN GET_AMINO_ACID IN TABLES.CPP");

112  }

122 formula_t*
123 get_formula_from_sequence(char* sequence)
124 {
125  formula_t* formula;
126  int size,i;
127  aa_CF_t aa;
128 
129  if(sequence==NULL)
130  {
131  error(USAGE_ERROR,"THE INDICATED IS EMPTY, NO FORMULA AVAILABLE",
132  "SEQUENCE IS NULL IN GET_FORMULA_FROM_SEQUENCE IN TABLES.CPP");
133  }
134 
135  size=strlen(sequence);
136  formula=copy_formula(get_util_formula("H"));
137  for(i=0;i<size;i++)
138  {
139  aa=get_amino_acid(sequence[i]);
140  add_formula(formula,aa.form);
141  add_formula(formula,get_util_formula("NHCHCO"));
142  }
143  add_formula(formula,get_util_formula("OH"));
144  return formula;
145 }

155 int
156 nextEntry (long *offset)
157 {
158  char c = ' ';
159  char last = ' ';
160  int nbMalloc = 1;
161  int count = -1;
162  long length = (long) 0;
163 
164  if(parsing_is_started==FALSE)
165  {
166  parsing_is_started=TRUE;
167  }
168 
169  c = getc (DbFile);
170  length = length + sizeof (c);
171 
172  if (c == EOF)
173  {
174  return FALSE;
175  }
176  if (c != '>')
177  {
178  fclose (DbFile);
179  error(EXECUTION_ERROR,"THE FASTA FILE DOES NOT RESPECT THE FASTA FORMAT",
180  "SYMBOL > WAITED BUT NOT FOUND IN NEXTENTRY IN SEQIO.CPP");
181  }
182 
183  free (SeqBuffer);
184  free (DescBuffer);
185 
186  SeqBuffer = (char *) malloc (sizeof (char) * MALLOC_STEP_SEQ);
187  DescBuffer = (char *) malloc (sizeof (char) * MALLOC_STEP_SEQ);
188 
189  /* fill description buffer */
190  while (c != EOF && (c != '\n'))
191  {
192  c = getc (DbFile);
193  length = length + sizeof (c);
194  if (c != '\n')
195  {
196  count++;
197  if (count >= (nbMalloc * MALLOC_STEP_SEQ - 1)) /* -1 for final \0 */
198  {
199  nbMalloc++;
200  DescBuffer = (char *)
201  realloc (DescBuffer,
202  nbMalloc * MALLOC_STEP_SEQ * sizeof (char));
203  }
204  DescBuffer[count] = c;
205  }
206  }
207  DescBuffer[count + 1] = '\0';
208 
209  c = getc (DbFile);
210  length = length + sizeof (c);
211  count = -1;
212  nbMalloc = 1;
213  /* fill sequence buffer */
214  while (c != EOF)
215  {
216  if (!(c == '>' && (last == '\n')))
217  {
218  /*if (c != '\n' && c != '\0') */
219  if (isalpha (c))
220  {
221  count++;
222  if (count >= (nbMalloc * MALLOC_STEP_SEQ - 1)) /* -1 for final \0 */
223  {
224  nbMalloc++;
225  SeqBuffer = (char *)
226  realloc (SeqBuffer,
227  nbMalloc * MALLOC_STEP_SEQ * sizeof (char));
228 
229  }
230  if (isalpha (c))
231  {
232  if (islower (c))
233  {
234  //SeqBuffer[count] = c;
235  SeqBuffer[count] = c + ('A'-'a');
236  }
237  else
238  {
239  //SeqBuffer[count] = c + 'a' - 'A';
240  SeqBuffer[count] = c;
241  }
242  //printf("%c", SeqBuffer[count]);
243  }
244  }
245  last = c;
246  c = getc (DbFile);
247  length = length + sizeof (char);
248  }
249  else
250  {
251  break;
252  }
253  }
254 
255  ungetc (c, DbFile);
256  length = length - sizeof (c);
257  *offset = Offset;
258  Offset += length;
259  SeqBuffer[count + 1] = '\0';
260 
261  return TRUE;
262 }

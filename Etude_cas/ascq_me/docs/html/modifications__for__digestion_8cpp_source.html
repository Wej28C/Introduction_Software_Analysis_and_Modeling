100 {

102  char* modif;
103 
104  lenght=strlen(configuration);
105  modif=NULL;
106  j=0;
107 
108  for(i=0;i<lenght;i++)
109  {

111  {
112  if(modif==NULL)

186 void
187 add_modification(int type,int id)
188 {
189  int i,already_added;/*,value;*/
190 
191  if((type!=FIXED)&&(type!=VARIABLE))
192  {
193  error(USAGE_ERROR,"THE TYPE OF THE WANTED MODIFICATION IS NOT CORRECT",
194  "TYPE MUST BE FIXE OR VARIABLE IN ADD_MODIFICATION IN MODIFICATIONS_FOR_DIGESTION.CPP");
195  }
196 
197  //because id is a natural value, 1 for the first, etc...
198  id--;
199 
200  if((id<0)&&(id>=get_total_number_of_modifications()))
201  {
202  error(USAGE_ERROR,"THE WANTED MODIFICATION IS NOT CORRECT",
203  "THE MODIFICATION HAS AN ODD VALUE IN ADD_MODIFICATION IN MODIFICATIONS_FOR_DIGESTION.CPP");
204  }
205 
206  already_added=FALSE;
207  i=0;
208  while((i<nb_modifications)&&(already_added==FALSE))
209  {
210  if(modifications[i].modification==id)
211  {
212  already_added=TRUE;
213  /*value=modifications[i].type;*/
214  }
215  i++;
216  }
217 
218  if(already_added==TRUE)
219  {
220  error(USAGE_ERROR,"THE WANTED MODIFICATION HAS BEEN ALREADY ADDED",
221  "A MODIFICATION CAN NOT BE ADDED TWO TIMES IN ADD_MODIFICATION IN MODIFICATIONS_FOR_DIGESTION.CPP");
222  }
223 
224  if(nb_modifications==0)
225  {
226  modifications=(post_translational_modification_t*)malloc(sizeof(post_translational_modification_t));
227  if(modifications==NULL)
228  {
229  error(MEMORY_ALLOCATION_ERROR,"A MEMORY ALLOCATION HAS FAILED FOR THE MODIFICATON TABLE",
230  "MODIFICATIONS IS NULL IN ADD_MODIFICATION IN MODIFICATIONS_FOR_DIGESTION.CPP");
231  }
232  }
233  else
234  {
235  modifications=(post_translational_modification_t*)realloc(modifications,
236  (nb_modifications+1)*sizeof(post_translational_modification_t));
237  if(modifications==NULL)
238  {
239  error(MEMORY_ALLOCATION_ERROR,"A MEMORY REALLOCATION HAS FAILED FOR THE MODIFICATON TABLE",
240  "MODIFICATIONS IS NULL IN ADD_MODIFICATION IN MODIFICATIONS_FOR_DIGESTION.CPP");
241  }
242  }
243 
244  modifications[nb_modifications].type=type;
245  modifications[nb_modifications].modification=id;
246 
247  nb_modifications++;
248 }

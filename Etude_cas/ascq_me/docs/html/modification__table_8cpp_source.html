192 int
193 can_be_used_on(int modif,char amino_acid,int place,int size,int first)
194 {
195  int i,length,good_side;
196 
197  if((modif<0)||(modif>=nb_total_modif))
198  {
199  error(USAGE_ERROR,"THE WANTED MODIF IS NOT CORRECT",
200  "MODIF HAS AN ODD VALUE IN CAN_BE_USED_ON IN MODIFICATIONS_TABLE.CPP");
201  }
202 
203  if(modifications_on_amino_acids[modif]==NULL)
204  {
205  if(side[modif]==C_TERM)
206  {
207  return place==(size-1)?TRUE:FALSE;
208  }
209  else if(side[modif]==N_TERM)
210  {
211  return place==0?TRUE:FALSE;
212  }
213  else if(side[modif]==BOTH_TERM)
214  {
215  return ((place==0)||(place==(size-1)))?TRUE:FALSE;
216  }
217  else if(side[modif]==PROTEIN)
218  {
219  return ((place==0)&&(first==TRUE))?TRUE:FALSE;
220  }
221  else
222  {
223  error(USAGE_ERROR,"THE WANTED MODIF CAN BE PUT ON ALL AMINO ACID, TOO DANGEROUS FOR THE EXECUTION TIME",
224  "PROBLEM FOR THE DESCRIPTION OF THE MODIF IN CAN_BE_USED IN MODIFICATIONS_TABLE.CPP");
225  //only for the compilation, the error stops the program.
226  return EXIT_FAILURE;
227  }
228  }
229  else
230  {
231  length=strlen(modifications_on_amino_acids[modif]);
232  if(side[modif]==C_TERM)
233  {
234  good_side=(place==(size-1)?TRUE:FALSE);
235  }
236  else if(side[modif]==N_TERM)
237  {
238  good_side=(place==0?TRUE:FALSE);
239  }
240  else if(side[modif]==BOTH_TERM)
241  {
242  good_side=(((place==0)||(place==(size-1)))?TRUE:FALSE);
243  }
244  else if(side[modif]==PROTEIN)
245  {
246  good_side=(((place==0)&&(first==TRUE))?TRUE:FALSE);
247  }
248  else
249  {
250  good_side=FALSE;
251  }
252 
253  if(side[modif]!=INDEPENDANT_TERM)
254  {
255  if(good_side==FALSE)
256  {
257  return FALSE;
258  }
259  }
260 
261  i=-1;
262  do
263  {
264  i++;
265  }
266  while((i<length)&&(modifications_on_amino_acids[modif][i]!=amino_acid));
267 
268  return i==length?FALSE:TRUE;
269  }

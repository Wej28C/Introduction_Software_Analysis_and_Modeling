129 void
130 load_configuration(char* configuration_file)
131 {
132  int read;
133  char buffer[50];
134  char keyword[50];
135 
136  FILE* file;
137 
138  configuration_file_name=(char*)malloc((strlen(configuration_file)+1)*sizeof(char));
139  if(configuration_file_name==NULL)
140  {
141  error(MEMORY_ALLOCATION_ERROR,"A MEMERY ALLOCATION HAS FAILED THE CONFIGURATION FILE NAME",
142  "CONFIGURATION_FILE_NAME IS NULL IN LOAD_CONFIGURATION IN ASCQ_ME_CONFIGURATION.CPP");
143  }
144  strcpy(configuration_file_name,configuration_file);
145 
146  read=0;
147  file=NULL;
148 
149  tables_initialisation();
150 
151  init_util_digestion();
152 
153  load_modifications(MODIFICATION_FILENAME);
154 
155  file=fopen(configuration_file,"r");
156  if(file==NULL)
157  {
158  error(IO_ERROR,"THE CONFIGURATION FILE DOES NOT EXIST OR IS NOT CORRECTLY OPENED",
159  "FILE IS NULL IN LOAD_CONFIGURATION IN ASCQ_ME_CONFIGURATION.CPP");
160  }
161 
162  do
163  {
164  jump_commentaries(file,'#');
165  read=fscanf(file,"%s",buffer);
166  if(read!=EOF)
167  {
168  strcpy(keyword,buffer);
169  }
170  read=fscanf(file,"%s",buffer);
171 
172  if(read!=EOF)
173  {
174  if(strcmp(buffer,"=")!=0)
175  {
176  error(USAGE_ERROR,"SYMBOL = WAITED AFTER THE KEYWORD IN THE CONFIGURATION FILE BUT NOT FOUND",
177  "BUFFER IS DIFFERENT FROM = IN LOAD_CONFIGURATION IN ASCQ_ME_CONFIGURATION.CPP");
178  }
179 
180  if((strcmp(keyword,"MIN_PEPTIDE_NB")==0)||
181  (strcmp(keyword,"MIN_MATCHED_PEPTIDE_NB")==0))
182  {
183  load_unsigned_int_keyword(file,keyword);
184  }
185  else if((strcmp(keyword,"MISSED_CLEAVAGES")==0)||
186  (strcmp(keyword,"ENZYME")==0)||
187  (strcmp(keyword,"NB_RESULTS")==0)||
188  (strcmp(keyword,"MAX_MASS")==0)||
189  (strcmp(keyword,"PEPTIDE_MAX_MASS")==0)||
190  (strcmp(keyword,"PEPTIDE_PRECISION")==0)||
191  (strcmp(keyword,"NB_C_ATOM")==0)||
192  (strcmp(keyword,"MAX_PEPTIDE_LENGTH")==0)||
193  (strcmp(keyword,"PRECISION")==0))
194  {
195  load_int_keyword(file,keyword);
196  }
197  else if((strcmp(keyword,"FULL_DIGEST")==0)||
198  (strcmp(keyword,"DECOY")==0)||
199  (strcmp(keyword,"FILE_TYPE")==0))
200  {
201  load_boolean_keyword(file,keyword);
202  }
203  else if((strcmp(keyword,"PEPTIDE_CHARGE")==0)||
204  (strcmp(keyword,"AMINO_ACID_B")==0)||
205  (strcmp(keyword,"AMINO_ACID_X")==0)||
206  (strcmp(keyword,"AMINO_ACID_Z")==0)||
207  (strcmp(keyword,"MODIFICATIONS")==0)||
208  (strcmp(keyword,"SPECIE")==0)||
209  (strcmp(keyword,"FASTA_FILE")==0)||
210  (strcmp(keyword,"SPECTRA_FILE")==0)||
211  (strcmp(keyword,"BASE_NAME")==0)||
212  (strcmp(keyword,"EXPERIMENTAL_SPECTRUM")==0))
213  {
214  load_string_keyword(file,keyword);
215  }
216  else if((strcmp(keyword,"GAUSSIAN")==0)||
217  (strcmp(keyword,"SENSITIVITY")==0)||
218  (strcmp(keyword,"EXP_MASS")==0)||
219  (strcmp(keyword,"MIN_EXP_MASS")==0)||
220  (strcmp(keyword,"MAX_EXP_MASS")==0)||
221  (strcmp(keyword,"THRESHOLD")==0))
222  {
223  load_float_keyword(file,keyword);
224  }
225  else
226  {
227  printf("%s = %d\n",keyword,(int)strlen(keyword));
228  error(USAGE_ERROR,"ONE OF THE KEYWORDS IN THE CONFIGURATION FILE IS NOT CORRECT",
229  "BUFFER HAS AN ODD VALUE IN LOAD_CONFIGURATION IN ASCQ_ME_CONFIGURATION.CPP");
230  }
231  }
232  }
233  while(read!=EOF);
234 
235  fclose(file);
236 
237  if(get_distribution_activation()==FALSE)
238  {
239  error(USAGE_ERROR,"NO EXPERIMENTAL SPECTRUM HAS BEEN LOADED",
240  "IS_DISTRIBUTION_INITIALISED IS FALSE IN LOAD_CONFIGURATION IN ASCQ_ME_CONFIGURATION.CPP");
241  }

137 aa_CF_t*
138 get_amino_acid_table(const char *filename,element_t *elem_table,int elem_table_size)
139 {
140  FILE *f;
141  char buffer[64];
142  int place;
143 
144  aa_CF_t *aa_table;
145  int i;
146 
147  int read;
148 
149  /* Allocate the table */
150  aa_table = (aa_CF_t *)malloc(sizeof(aa_CF_t)*26);
151  if(aa_table==NULL)
152  {
153  error(MEMORY_ALLOCATION_ERROR,"A MEMORY ALLOCATION HAS FAILED FOR THE AMINO ACID TABLE",
154  "AA_TABLE IS NULL IN GET_AMINO_ACID_TABLLE IN AA.CPP");
155  }
156  for(i=0;i<26;i++)
157  {
158  aa_table[i].form=NULL;
159  aa_table[i].add_modif=NULL;
160  aa_table[i].sub_modif=NULL;
161  }
162 
163  f = fopen (filename, "r");
164 
165  if (f==NULL)
166  {
167  error(IO_ERROR,"THE AMINO ACID FILE CAN NOT BE CORRECTLY OPENED",
168  "FILE IS NULL IN GET_AMINO_ACID_TABLLE IN AA.CPP");
169  }
170  else
171  {
172  /*------------------------------------------------------------------------------------*/
173  /*for the unix version */
174  read=fscanf(f,"%s",buffer);
175  while(read!=-1)
176  {
177  place=(int)(toupper(buffer[0]-'A'));
178  read=fscanf(f,"%s",buffer);
179  if(read!=-1)
180  {
181  aa_table[place].form=read_formula(buffer,elem_table,elem_table_size);
182  read=fscanf(f,"%s",buffer);
183  }
184  if(read!=-1)
185  {
186  aa_table[place].add_modif=read_formula(buffer,elem_table,elem_table_size);
187  read=fscanf(f,"%s",buffer);
188  }
189  if(read!=-1)
190  {
191  aa_table[place].add_modif=read_formula(buffer,elem_table,elem_table_size);
192  read=fscanf(f,"%s",buffer);
193  }
194  }
195  }
196  fclose(f);
197 
198  place=(int)('X'-'A');
199  aa_table[place].form=copy_formula(aa_table[X-'A'].form);
200  place=(int)('Z'-'A');
201  aa_table[place].form=copy_formula(aa_table[Z-'A'].form);
202  place=(int)('B'-'A');
203  aa_table[place].form=copy_formula(aa_table[B-'A'].form);
204  return aa_table;
205 }

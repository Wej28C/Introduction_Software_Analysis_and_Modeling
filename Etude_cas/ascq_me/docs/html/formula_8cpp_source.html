89 int
90 closest_int(double mass)
91 {
92  int up,down;
93  down=(int)mass;
94  up=down+1;
95 
96  if((mass-(double)down)<((double)up-mass))
97  {
98  return down;
99  }
100  return up;

175 formula_t*
176 read_formula(const char *c,element_t *elem_table,int elem_table_size)
177 {
178  int current_quantity,j,i,isotop,length;
179  char atom_symbol[3];
180  char quantity_buffer[16];
181  formula_t *form;
182 
183  /*---------------------------------------------------*/
184  /*unix version */
185  /* for '*' case */
186  if(c[0]=='*')
187  {
188  return NULL;
189  }
190  /*----------------------------------------------------*/
191 
192 
193  /* allocate formulat structure */
194  form = (formula_t *) malloc(sizeof(formula_t));
195  if(form==NULL)
196  {
197  error(MEMORY_ALLOCATION_ERROR,"A MEMORY ALLOCATION HAS FAILED DURING THE READING OF A FORMULA",
198  "FORM IS NULL IN READ_FORMULA IN FORMULA.CPP");
199  }
200 
201  /* add 1 composition (ex : H1) */
202  form->nb_comp = 1;
203  form->comps = (composition_t *) malloc(sizeof(composition_t));
204  if(form->comps==NULL)
205  {
206  error(MEMORY_ALLOCATION_ERROR,"A MEMORY ALLOCATION HAS FAILED DURING A FORMULA READING",
207  "FORM->COMPS IS NULL IN READ_FORMULA IN FORMULA.CPP");
208  }
209  length=strlen(c);
210  i=0;/* index on c */
211  while(i<length)
212  {
213  /* reading symbol */
214  current_quantity=1; /* for implicite quantity ex:CH3*/
215  atom_symbol[1] =atom_symbol[2]='\0';
216  atom_symbol[0] = c[i];
217  i++;
218  if(i<length)
219  {
220  if(c[i]>='a' && c[i]<='z')
221  {
222  atom_symbol[1] = c[i];
223  i++;
224  }
225  }
226  isotop=-1;
227  if(i<length)
228  {
229  /* reading quantity */
230  if(c[i]=='(')
231  {
232  i++;
233  j=0;
234  while((i<length)&&(isdigit(c[i])))
235  {
236  quantity_buffer[j]=c[i];
237  j++;
238  i++;
239  }
240  quantity_buffer[j]='\0';
241  if(quantity_buffer[0]!='\0')
242  {
243  isotop=atoi(quantity_buffer);
244  }
245  if(i<length)
246  {
247  if(c[i]!=')')
248  {
249  error(USAGE_ERROR,"A READ FORMULA HAS NOT THE RIGHT FORM",
250  "ISOTOP WANTED BUT THE ')' MISSES IN READ_FORMULA IN FORMULA.CPP");
251  }
252  i++;
253  }
254  }
255  j=0;
256  while((i<length)&&(isdigit(c[i])))
257  {
258  quantity_buffer[j]=c[i];
259  i++;
260  j++;
261  }
262  quantity_buffer[j]='\0';
263  if(quantity_buffer[0]!='\0')
264  {
265  current_quantity=atoi(quantity_buffer);
266  }
267 
268 
269  /* Affect the element_t in the result */
270  if(affect_atom(atom_symbol,current_quantity,isotop,&form->comps[form->nb_comp-1],elem_table,elem_table_size)==FALSE)
271  {
272  return NULL;
273  }
274 
275  if(i<length) /* Read the next element in the formula, if exists */
276  {
277  form->nb_comp++; /* allocate new composition */
278  form->comps = (composition_t *) realloc(form->comps,sizeof(composition_t)*(form->nb_comp+1));
279  }
280  }
281  else
282  {
283  if(affect_atom(atom_symbol,current_quantity,isotop,&form->comps[form->nb_comp-1],elem_table,elem_table_size)==FALSE)
284  {
285  return NULL;
286  }
287  }
288  }
289 
290  return form;
291 }

313

319  }

321  attribute=read_until(f,'=');
322  read=fscanf(f,"%c",&c);

324  {

326  {
327  error(EXECUTION_ERROR,"THE PARSING OF THE ELEMENT TABLE FILE HAS FAILED","READING UNTIL = BUT NEXT SYMBOL IS NOT =");
328  }
329  }
330 
331  while(attribute!=NULL)
332  {
333  if(strncmp(attribute,"Atomic Symbol",strlen("Atomic Symbol"))==0) /* match atomic symbol */
334  {
335  value=read_line(f);
336  /*****************************************************/
337  /* new portion in order to delete parasite character
338  * for example : due to dos text format ==> ^M
339  * allow to give the atom weight file in unix or dos format
340  * if the file is in unix format ==> this portion do nothing */
341  i=0;
342  while((i<(int)strlen(value))&&(isalpha(value[i])))
343  {
344  i++;
345  }
346  if(i!=(int)strlen(value))
347  {
348  value[i]='\0';
349  }
350  /*****************************************************/
351 
352  /* new symbol */
353  if(index==-1 || strcmp(value,elem_table[index].symbol)!=0)
354  {
355  index++;
356  if(index>0) /* realloc new elem_table*/
357  {
358  elem_table= (element_t *) realloc(elem_table,sizeof(element_t)*(index+2));
359  if(elem_table==NULL)
360  {
361  error(MEMORY_ALLOCATION_ERROR,"A MEMORY REALLOCATION HAS FAILED DURING THE READING",
362  "ELEM_TABLE IS NULL IN GET_ELEMENT_TABLE IN FORMULA.CPP\n");
363  }
364  }
365 
366  strcpy(elem_table[index].symbol,value);
367 
368  elem_table[index].nb_isotops=0;
369  elem_table[index].isotops=(isotop_t *) malloc(sizeof(isotop_t));
370  }
371 
372  free(value);
373  }
374  /* match relative .. */
375  else if(strncmp(attribute,"Relative Atomic Mass",strlen("Relative Atomic Mass"))==0)
376  {
377  value=read_until(f,'(');
378  jump_a_line(f);
379  elem_table[index].nb_isotops++;
380  /* realloc new isotop */
381  if(elem_table[index].nb_isotops>1)
382  {
383  elem_table[index].isotops = (isotop_t *) realloc(elem_table[index].isotops,sizeof(isotop_t)*(elem_table[index].nb_isotops));
384  }
385 
386  if(value!=NULL)
387  {
388  /* associate value */
389  elem_table[index].isotops[elem_table[index].nb_isotops-1].relative_mass=(double) atof(value); /* for precision */
390  }
391  else
392  {
393  elem_table[index].isotops[elem_table[index].nb_isotops-1].relative_mass=0.0;
394  }
396  elem_table[index].isotops[elem_table[index].nb_isotops-1].isotopic_composition = 0.0; /* by default*/
398 
399  free(value);
400  }
401 
402  else if(strncmp(attribute,"Isotopic Composition",strlen("Isotopic Composition"))==0) /* match isotopic.. */
403  {
404  value=read_line(f);
405  /* associate value */
406  if(value!=NULL)
407  {
408  elem_table[index].isotops[elem_table[index].nb_isotops-1].isotopic_composition=(double) atof(value);/* for precision */
409  }
410  free(value);
411  }
412  else
413  {
414  jump_a_line(f);
415  }
416 
417  free(attribute);
418  jump_commentaries(f,'/');
419  attribute=read_until(f,'=');
420  read=fscanf(f,"%c",&c);
421  if(read!=EOF)
422  {
423  if(c!='=')
424  {
425  error(EXECUTION_ERROR,"THE PARSING OF THE ELEMENT TABLE FILE HAS FAILED","READING UNTIL = BUT NEXT SYMBOL IS NOT =");
426  }
427  }
428  }
429  (*nb_elements) = index+1;
430 
431  fclose(f);
432 
433  return elem_table;
434 }
